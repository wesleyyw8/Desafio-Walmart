
CREATE DATABASE WALMARTDESAFIO;
USE WALMARTDESAFIO;

CREATE TABLE CLIENTE (
	id INT NOT NULL AUTO_INCREMENT,
	nome VARCHAR(255) NOT NULL ,
	idade INT,
	sexo VARCHAR(1),
	PRIMARY KEY (id)
);

CREATE TABLE ENDERECO(
	id INT NOT NULL AUTO_INCREMENT,
	nome VARCHAR(255) NOT NULL,
	CEP INT,
	CIDADE VARCHAR(255) NOT NULL,
	PAIS VARCHAR(255) NOT NULL,
	PRIMARY KEY (id)
);

CREATE TABLE PRODUTO(
	id INT NOT NULL AUTO_INCREMENT,
	nome VARCHAR(255) NOT NULL,
	tipo VARCHAR(255),
	preco DOUBLE,
	quantidade INT,
	PRIMARY KEY (id)
);
